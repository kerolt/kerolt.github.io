class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        stack<int> st;
        int res = 0;
        heights.push_back(-1);
        for (int i = 0; i < heights.size(); ++i) {
            while (!st.empty() && heights[i] < heights[st.top()]) {
                int h = heights[st.top()];
                st.pop();
                int w = i; // å³ w = i - (-1) - 1
                if (!st.empty()) {
                    w = i - st.top() - 1;
                }
                res = max(res, h * w);
            }
            st.push(i);
        }
        return res;
    }
};
