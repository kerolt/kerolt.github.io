bool IOManager::AddEvent(int fd, Event event, std::function<void()> cb) {

    {
        std::shared_lock rw_lock(rw_mutex_);
        if (fd_contexts_.size() > fd) {
            fd_ctx = fd_contexts_[fd];
            rw_lock.unlock();
